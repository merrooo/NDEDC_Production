    function angleToRange(theta) {
        if (isNaN(theta)) return "undefined";
        const v = theta;
        if (v > 120 && v <= 180) return "180 > theta > 120";
        if (v > 60 && v <= 120) return "120 > theta > 60";
        if (v > 0 && v <= 60) return "0 < theta < 60";
        if (v >= -60 && v <= 0) return "0 > theta > -60";
        if (v >= -120 && v < -60) return "-60 > theta > -120";
        if (v >= -180 && v < -120) return "-120 > theta > -180";
        return "out of range";
    }

    // ==============================================================
    // PHASOR DIAGRAM: COUNTER-CLOCKWISE ROTATION
    // CURRENT LAGS VOLTAGE → Current is CLOCKWISE (negative shift)
    // ==============================================================
    function drawPhasorDiagram(thetaV_R, thetaV_Y, thetaV_B, lagDegrees = 30) {
        const canvas = document.getElementById('phasorCanvas');
        const ctx = canvas.getContext('2d');
        const size = canvas.width = canvas.height = 500;
        const centerX = size/2, centerY = size/2;
        const radius = size * 0.35;
        ctx.clearRect(0, 0, size, size);
        
        // Draw outer circle
        ctx.beginPath();
        ctx.arc(centerX, centerY, radius + 10, 0, 2 * Math.PI);
        ctx.strokeStyle = "#cbd5e1";
        ctx.setLineDash([5, 8]);
        ctx.stroke();
        ctx.setLineDash([]);
        
        // Draw grid lines and angle labels
        for(let i=0; i<360; i+=30) {
            let rad = i * Math.PI/180;
            let x1 = centerX + (radius-10) * Math.cos(rad);
            let y1 = centerY + (radius-10) * Math.sin(rad);
            let x2 = centerX + (radius+10) * Math.cos(rad);
            let y2 = centerY + (radius+10) * Math.sin(rad);
            ctx.beginPath();
            ctx.moveTo(x1,y1);
            ctx.lineTo(x2,y2);
            ctx.strokeStyle = "#e2e8f0";
            ctx.stroke();
            
            // Add angle labels
            ctx.font = "9px 'Inter'";
            ctx.fillStyle = "#9ca3af";
            if (i === 0) ctx.fillText("0°", centerX + radius + 12, centerY - 3);
            else if (i === 90) ctx.fillText("90°", centerX + 3, centerY - radius - 8);
            else if (i === 180) ctx.fillText("180°", centerX - radius - 28, centerY - 3);
            else if (i === 270) ctx.fillText("-90°", centerX + 3, centerY + radius + 18);
            else if (i === 60) ctx.fillText("60°", centerX + radius * 0.55, centerY - radius * 0.85);
            else if (i === 120) ctx.fillText("120°", centerX - radius * 0.55, centerY - radius * 0.85);
            else if (i === 240) ctx.fillText("-120°", centerX - radius * 0.55, centerY + radius * 0.85);
            else if (i === 300) ctx.fillText("-60°", centerX + radius * 0.55, centerY + radius * 0.85);
        }
        
        function drawPhasor(angleDeg, color, lineWidth, dashed, label) {
            let rad = (angleDeg * Math.PI) / 180;
            let x = centerX + radius * Math.cos(rad);
            let y = centerY + radius * Math.sin(rad);
            
            ctx.beginPath();
            ctx.moveTo(centerX, centerY);
            ctx.lineTo(x, y);
            ctx.strokeStyle = color;
            ctx.lineWidth = lineWidth;
            ctx.setLineDash(dashed ? [8, 6] : []);
            ctx.stroke();
            ctx.setLineDash([]);
            
            // Arrowhead
            let arrowRad = rad;
            let arrX = x - 12 * Math.cos(arrowRad);
            let arrY = y - 12 * Math.sin(arrowRad);
            ctx.beginPath();
            ctx.moveTo(x, y);
            ctx.lineTo(arrX - 5 * Math.sin(arrowRad), arrY + 5 * Math.cos(arrowRad));
            ctx.lineTo(arrX + 5 * Math.sin(arrowRad), arrY - 5 * Math.cos(arrowRad));
            ctx.fillStyle = color;
            ctx.fill();
            
            // Label
            ctx.font = "bold 11px 'Inter'";
            ctx.fillStyle = color;
            ctx.fillText(label, x + 10 * Math.cos(rad), y + 10 * Math.sin(rad));
        }
        
        // Draw VOLTAGE phasors (Solid)
        if(!isNaN(thetaV_R)) drawPhasor(thetaV_R, "#dc2626", 3, false, "V_R");
        if(!isNaN(thetaV_Y)) drawPhasor(thetaV_Y, "#d97706", 3, false, "V_Y");
        if(!isNaN(thetaV_B)) drawPhasor(thetaV_B, "#2563eb", 3, false, "V_B");
        
        // Draw CURRENT phasors (Dashed) - LAG = CLOCKWISE = subtract angle
        const lag = -lagDegrees;
        if(!isNaN(thetaV_R)) drawPhasor(thetaV_R + lag, "#f97316", 2.2, true, "I_R");
        if(!isNaN(thetaV_Y)) drawPhasor(thetaV_Y + lag, "#f97316", 2.2, true, "I_Y");
        if(!isNaN(thetaV_B)) drawPhasor(thetaV_B + lag, "#f97316", 2.2, true, "I_B");
        
        // Center point
        ctx.fillStyle = "#334155";
        ctx.beginPath();
        ctx.arc(centerX, centerY, 5, 0, 2 * Math.PI);
        ctx.fill();
        
        // CCW arrow indicator
        ctx.beginPath();
        let arrowRad = -100 * Math.PI/180;
        let arrowX = centerX + (radius + 25) * Math.cos(arrowRad);
        let arrowY = centerY + (radius + 25) * Math.sin(arrowRad);
        ctx.moveTo(arrowX, arrowY);
        ctx.lineTo(arrowX - 8, arrowY - 5);
        ctx.lineTo(arrowX - 8, arrowY + 5);
        ctx.fillStyle = "#9ca3af";
        ctx.fill();
        ctx.fillStyle = "#9ca3af";
        ctx.font = "9px 'Inter'";
        ctx.fillText("CCW", arrowX - 28, arrowY - 2);
    }
    
    function performDiagnostic() {
        const rawR = parseFloat(document.getElementById('thetaR').value);
        const rawY = parseFloat(document.getElementById('thetaY').value);
        const rawB = parseFloat(document.getElementById('thetaB').value);
        
        if (isNaN(rawR) || isNaN(rawY) || isNaN(rawB)) {
            document.getElementById('resultContainer').style.display = "block";
            document.getElementById('dynamicStatus').innerHTML = "⚠️ INCOMPLETE DATA";
            document.getElementById('remediationSteps').innerHTML = "<div class='action-step'>Enter valid angles for all three phases.</div>";
            return;
        }
        
        const rangeR = angleToRange(rawR);
        const rangeY = angleToRange(rawY);
        const rangeB = angleToRange(rawB);

            entry.r[0] === rangeR && entry.r[1] === rangeY && entry.r[2] === rangeB
        );
        
        const resultDiv = document.getElementById('resultContainer');
        resultDiv.style.display = "block";
        const statusChip = document.getElementById('dynamicStatus');
        const voltRefDiv = document.getElementById('voltageRef');
        const stepsContainer = document.getElementById('remediationSteps');
        
        if (!matchedEntry) {
            statusChip.className = "status-chip fault";
            statusChip.innerHTML = "❌ UNKNOWN ANGLE CONFIGURATION";
            voltRefDiv.innerHTML = `⚠️ No match for R:${rangeR}, Y:${rangeY}, B:${rangeB}`;
            stepsContainer.innerHTML = "<div class='action-step'>Angle combination not found in database.</div>";
            drawPhasorDiagram(rawR, rawY, rawB, 30);
            document.getElementById('vectorInsight').innerHTML = `⚠️ Unrecognized: ${rawR}°, ${rawY}°, ${rawB}°`;
            return;
        }
        
        const optimalTerminals = ["1-3", "4-6", "7-9"];
        const isWiringCorrect = matchedEntry.c.every((term, idx) => term === optimalTerminals[idx]);
        
        const iR = rawR - 30;
        const iY = rawY - 30;
        const iB = rawB - 30;
